### 文件指针
```

int main() {
	FILE *pFile;//声明一个文件指针
	char *szReadTextBuffer;// 申明一个char类型的指针，用来存储读进来的东西
	int nReadFileSize;//读取到文件的尺寸
	int nReadRetSize;//声明读取到文件的一个真实长度
					  //r 读取 w写出 b二进制
	pFile = fopen("C:\\Users\\rkvir\\Desktop\\123.txt", "rb");//二进制读取文件内容 赋值给pFile指针
	if (pFile == NULL) {//判断如果指针是一个空值则读取文件失败
		printf("Open File Failed!");
		exit(0);//推出进程
	}
	fseek(pFile, 0, SEEK_END);//文件指针设置到末尾，文件指针在末尾处就知道文件的长度 
	nReadFileSize = ftell(pFile);//ftell判断文件指针位置返回一个大小值
	rewind(pFile);//文件指针复位
	szReadTextBuffer = (char *)malloc((nReadFileSize * sizeof(char) + 1));//申请一块内存给读取进来的内容,因为末尾需要00结尾所以加1 
	if (szReadTextBuffer == NULL) {//判断szReadTextBuffer指针是否为空值，是则申请内存失败退出
		printf("申请内存失败");
		exit(0);
	}
	memset(szReadTextBuffer, 0, nReadFileSize + 1);//给申请的内存赋初始值 0
	nReadRetSize = fread(szReadTextBuffer, 1, nReadFileSize, pFile);//fread文件读取，szReadTextBuffer用来存储读取的数据，1代表一位一位的取值，nReadFileSize长度，从pFile指针里面取
	if (nReadRetSize != nReadFileSize) {//判断如果nReadFileSize
		printf("文件读取失败");
		exit(0);
	}
	puts(szReadTextBuffer);
	fclose(pFile);//资源释放
	return 0;
}
```
