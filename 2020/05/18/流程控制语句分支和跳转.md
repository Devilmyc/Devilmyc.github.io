```

    printf("dui");

```
//?:
```
int a =1;
int b=2;
int c;
c = (表达式)?7:5;//表达式为true 则c的值为问好后面的，为假则为冒号后面的值；

//continue 跳出循环
```
if(a == 3){
    continue; //当条件符合时，continue就跳转本次循环执行下一次的
    break；//当条件符合时退出循环
}
printf();
```

```
int flag =5;

    case 1;

        printf("2");
        break;
    case 3;
        printf("3");
        break;
    case 4;
        printf("4");
        break;
    case 5;
        printf("5");
        break;
    default: //如果参数都没有匹配到case则进入default

        break;
