## 结构体
```
int main() {
	//Gmae Player Nmae HP MP 坐标xyz
	struct GmaePlayerInfo { //struct后跟结构体的名字
		char PlayerName[50];
		int HP;
		int MP;
		float x;
		float y;
		float z;
	};
	struct NPCInfo { //struct后跟结构体的名字
		char NPCNmae[50];
		int HP;
		int MP;
		float x;
		float y;
		float z;
	};
	struct GmaePlayerInfo rkvir = { "rkvir",500,500,123.55,155.55,0 };
	struct GmaePlayerInfo myc = { .PlayerName = "myc",.MP = 1000 };
	struct NPCInfo MyNPCinfo[50];
	for (size_t i = 0; i <50; i++) {
		strcpy(MyNPCinfo[i].NPCNmae, "NPC");
		MyNPCinfo[i].HP = rand();
		MyNPCinfo[i].MP = rand();
	}
	printf("请设置游戏名");
	scanf("%s", myc.PlayerName);


	return 0;
}
```
